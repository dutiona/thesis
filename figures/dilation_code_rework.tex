template <Image In, StructuringElement SE,
          WritableImage Out>
void dilate(const In& ima, const SE& se,
            Out& out){
  for(auto pnt : ima.points()) {
    out(pnt) = ima(pnt);
    for(auto nbx : se(pnt))
      out(pnt) = |\colorbox{red!50!white}{max}|(out(pnt), img(nx));
  }
}
\end{minted}
