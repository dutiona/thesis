\begin{minted}[highlightlines=5,highlightcolor=yellow!60!white]{python}
def dilate(img, se, out):
  for pnt in img.points():
    out(pnt) = img(pnt)
    for nx in se(pnt):
      out(pnt) = max(out(pnt), img(nx))
\end{minted}
