\begin{minted}[highlightlines=8,highlightcolor=yellow!60!white]{c++}
template <Image In, StructuringElement SE,
          WritableImage Out>
void erode(const In& ima, const SE& se,
            Out& out){
  for(auto pnt : ima.points()) {
    out(pnt) = ima(pnt);
    for(auto nbx : se(pnt))
      out(pnt) = min(out(pnt), img(nx));
  }
}
\end{minted}
